    <script>

        // Navigation - Abschnitte wechseln

        const navLinks = document.querySelectorAll('nav a');

        const sections = document.querySelectorAll('main section');


        navLinks.forEach(link => {

            link.addEventListener('click', (e) => {

                e.preventDefault();

                const target = link.getAttribute('data-section');


                sections.forEach(section => {

                    section.classList.toggle('active', section.id === target);

                });


                // Reset messages & errors

                document.querySelectorAll('p[id$="Message"]').forEach(el => el.style.display = 'none');

                document.querySelectorAll('.error-message').forEach(el => el.style.display = 'none');

            });

        });


        // Einfacher Login-Simulation (nur Beispiel)

        document.getElementById('loginForm').addEventListener('submit', e => {

            e.preventDefault();

            const username = e.target.username.value.trim();

            const password = e.target.password.value.trim();

            const errorMsg = document.getElementById('loginError');


            // Beispiel: Benutzername "mieter", Passwort "1234"

            if(username === 'mieter' && password === '1234') {

                alert('Login erfolgreich! Willkommen ' + username);

                errorMsg.style.display = 'none';

                // Nach Login Startseite anzeigen (oder andere Logik)

                sections.forEach(section => {

                    section.classList.toggle('active', section.id === 'home');

                });

            } else {

                errorMsg.style.display = 'block';

            }

        });


        // Schaden melden Formular

        document.getElementById('schadenForm').addEventListener('submit', e => {

            e.preventDefault();

            // Hier könntest du die Daten per fetch zu einem Server senden

            e.target.reset();

            document.getElementById('schadenMessage').style.display = 'block';

        });
